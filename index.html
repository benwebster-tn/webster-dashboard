/** * HealthSnapshot.gs * * Server-side function for the expanded Agency Health Card. * Returns pre-computed retention (overall + by carrier), leakage stats, * and loss budget tracking. * * Called via RPC from the standalone dashboard: * fetchGS("actionHealthSnapshot", function(r) { ... }); * * Sheet IDs (from Code.gs constants): * KNOWN_SHEETS.bob = '1x7QRQMulDtt9H6J9dIowxE5IT36cmEHsRyZTcgCIF6Y' * KNOWN_SHEETS.renewals = '1Nb9w1u9dPYcEeaGAcu3yGRm1zSvdG8kuVhkfP5ROiP4' */ function actionHealthSnapshot() { try { var today = new Date(); var currentMonth = today.getMonth(); // 0-based (0=JAN) // ─── 1. RENEWAL RETENTION (overall + by carrier) ────────────────── var monthNames = ['JAN', 'FEB', 'MAR', 'APR', 'MAY', 'JUN', 'JUL', 'AUG', 'SEP', 'OCT', 'NOV', 'DEC']; var ss = SpreadsheetApp.openById(KNOWN_SHEETS.renewals); // Carrier buckets — Erie, Travelers, Progressive, Broker, Other var TRACKED_CARRIERS = ['ERIE', 'TRAVELERS', 'PROGRESSIVE', 'BROKER']; var overallUp = 0; var overallRenewed = 0; var carrierStats = {}; TRACKED_CARRIERS.forEach(function(c) { carrierStats[c] = { premiumUp: 0, premiumRenewed: 0, clients: 0, renewedClients: 0, policies: [] }; }); carrierStats['OTHER'] = { premiumUp: 0, premiumRenewed: 0, clients: 0, renewedClients: 0, policies: [] }; // Per-month retention for trend var monthlyRetention = []; for (var m = 0; m <= currentMonth && m < monthNames.length; m++) { var sheet; try { sheet = ss.getSheetByName(monthNames[m]); } catch (e) { continue; } if (!sheet) continue; var lastRow = sheet.getLastRow(); if (lastRow < 6) continue; var lastCol = sheet.getLastColumn(); var data = sheet.getRange(6, 1, Math.min(lastRow - 5, 95), lastCol).getValues(); var mUp = 0, mRenewed = 0; for (var i = 0; i < data.length; i++) { var row = data[i]; var name = row[0] ? String(row[0]).trim() : ''; if (!name) continue; var carrier = row[1] ? String(row[1]).trim().toUpperCase() : ''; var premUp = parseFloat(row[3]) || 0; var isPaid = (row[4] ? String(row[4]).trim() : '').toUpperCase() === 'YES'; var premNew = parseFloat(row[6]) || 0; var notes = row[12] ? String(row[12]).trim().toUpperCase() : ''; var isCancelled = notes.indexOf('CANCEL') !== -1; // Skip cancelled policies from "up for renewal" totals if (isCancelled) continue; // Determine carrier bucket var bucket = 'OTHER'; for (var c = 0; c < TRACKED_CARRIERS.length; c++) { if (carrier.indexOf(TRACKED_CARRIERS[c]) !== -1) { bucket = TRACKED_CARRIERS[c]; break; } } // Overall overallUp += premUp; carrierStats[bucket].premiumUp += premUp; carrierStats[bucket].clients++; if (isPaid) { overallRenewed += premNew; carrierStats[bucket].premiumRenewed += premNew; carrierStats[bucket].renewedClients++; } // Track individual policy for carrier drill-down carrierStats[bucket].policies.push({ name: name, month: monthNames[m], premiumUp: premUp, premiumNew: isPaid ? premNew : 0, paid: isPaid, carrier: carrier }); // Monthly tracking mUp += premUp; if (isPaid) mRenewed += premNew; } monthlyRetention.push({ month: monthNames[m], premiumUp: mUp, premiumRenewed: mRenewed, retention: mUp > 0 ? Math.round((mRenewed / mUp) * 1000) / 10 : 0 }); } var overallRetention = overallUp > 0 ? Math.round((overallRenewed / overallUp) * 1000) / 10 : 0; // ─── 2. LEAKAGE (from Cancellations 2026 tab on BOB sheet) ──────── var cancSheet; var cancTotal = 0, nrTotal = 0, cancCount = 0, nrCount = 0; try { cancSheet = SpreadsheetApp.openById(KNOWN_SHEETS.bob).getSheetByName('Cancellations 2026'); } catch (e) { cancSheet = null; } // Track cancellation losses by carrier for true retention calc var cancByCarrier = {}; if (cancSheet) { var cancLastRow = cancSheet.getLastRow(); if (cancLastRow > 3) { var cancLastCol = cancSheet.getLastColumn(); // Row 1-2 = titles/totals, Row 3 = headers, Row 4+ = data var cancData = cancSheet.getRange(4, 1, cancLastRow - 3, Math.min(cancLastCol, 11)).getValues(); // Actual column layout (from CancellationBridge_v2): // A=Name, B=Carrier, C=Type, D=Policy#, E=Date, // F=Old Premium, G=Cancellation Premium, H=Non-Renewal Premium, // I=Reason, J=Notes // G (index 6) = premium lost via cancellation // H (index 7) = premium lost via non-renewal var MONTHS_LIST = ['JANUARY','FEBRUARY','MARCH','APRIL','MAY','JUNE', 'JULY','AUGUST','SEPTEMBER','OCTOBER','NOVEMBER','DECEMBER']; for (var r = 0; r < cancData.length; r++) { var cRow = cancData[r]; var cName = cRow[0] ? String(cRow[0]).trim() : ''; if (!cName) continue; // Skip month divider rows and section headers var cNameUpper = cName.toUpperCase(); if (MONTHS_LIST.indexOf(cNameUpper) !== -1 || cNameUpper === 'PERSONAL' || cNameUpper === 'COMMERCIAL') continue; // Carrier from column B (index 1) var cCarrier = cRow[1] ? String(cRow[1]).trim().toUpperCase() : ''; var cBucket = 'OTHER'; for (var cb = 0; cb < TRACKED_CARRIERS.length; cb++) { if (cCarrier.indexOf(TRACKED_CARRIERS[cb]) !== -1) { cBucket = TRACKED_CARRIERS[cb]; break; } } // Cancellation premium = column G (index 6) var cancPrem = 0; if (typeof cRow[6] === 'number') { cancPrem = cRow[6]; } else if (cRow[6]) { cancPrem = parseFloat(String(cRow[6]).replace(/[$,]/g, '')) || 0; } // Non-renewal premium = column H (index 7) var nrPrem = 0; if (typeof cRow[7] === 'number') { nrPrem = cRow[7]; } else if (cRow[7]) { nrPrem = parseFloat(String(cRow[7]).replace(/[$,]/g, '')) || 0; } var lostPrem = cancPrem + nrPrem; // Track by carrier if (!cancByCarrier[cBucket]) { cancByCarrier[cBucket] = { premium: 0, count: 0 }; } cancByCarrier[cBucket].premium += lostPrem; cancByCarrier[cBucket].count++; if (nrPrem > 0) { nrTotal += nrPrem; nrCount++; } else if (cancPrem > 0) { cancTotal += cancPrem; cancCount++; } else { // Row has a name but no premium in G or H — still count it cancCount++; } } } } var totalLeakage = cancTotal + nrTotal; // ─── 3. BUILD CARRIER ARRAY (after cancellation data is ready) ──── // IMPORTANT: This must come AFTER the cancellation loop so cancByCarrier is populated var carrierArray = []; var allKeys = TRACKED_CARRIERS.concat(['OTHER']); for (var k = 0; k < allKeys.length; k++) { var key = allKeys[k]; var cs = carrierStats[key]; if (cs.clients === 0) continue; var cLoss = cancByCarrier[key] || { premium: 0, count: 0 }; var renewalRetention = cs.premiumUp > 0 ? Math.round((cs.premiumRenewed / cs.premiumUp) * 1000) / 10 : 0; var totalAtRisk = cs.premiumUp + cLoss.premium; var trueRetention = totalAtRisk > 0 ? Math.round((cs.premiumRenewed / totalAtRisk) * 1000) / 10 : 0; carrierArray.push({ name: key.charAt(0) + key.slice(1).toLowerCase(), premiumUp: cs.premiumUp, premiumRenewed: cs.premiumRenewed, clients: cs.clients, renewedClients: cs.renewedClients, retention: renewalRetention, cancLostPremium: cLoss.premium, cancLostCount: cLoss.count, totalAtRisk: totalAtRisk, trueRetention: trueRetention, policies: cs.policies }); } // ─── 4. BOB TOTAL (for leakage % calc) ──────────────────────────── var bobSheet = SpreadsheetApp.openById(KNOWN_SHEETS.bob).getSheetByName('BOB'); var bobLastRow = bobSheet.getLastRow(); var bobTotal = 0; var bobPolicies = 0; if (bobLastRow > 1) { var bobData = bobSheet.getRange(2, 1, bobLastRow - 1, bobSheet.getLastColumn()).getValues(); for (var b = 0; b < bobData.length; b++) { var bName = bobData[b][0] ? String(bobData[b][0]).trim() : ''; if (!bName) continue; var bPrem = 0; if (typeof bobData[b][6] === 'number') { bPrem = bobData[b][6]; } else if (typeof bobData[b][6] === 'string') { bPrem = parseFloat(String(bobData[b][6]).replace(/[$,]/g, '')) || 0; } bobTotal += bPrem; bobPolicies++; } } var leakagePctBob = bobTotal > 0 ? Math.round((totalLeakage / bobTotal) * 10000) / 100 : 0; // ─── 5. LOSS BUDGET ─────────────────────────────────────────────── var lossBudget = 389000; // 2026 annual loss budget from KPIs return { ok: true, data: { retention: { overall: overallRetention, trueOverall: (overallUp + totalLeakage) > 0 ? Math.round((overallRenewed / (overallUp + totalLeakage)) * 1000) / 10 : 0, goal: 92, premiumUp: overallUp, premiumRenewed: overallRenewed, totalAtRisk: overallUp + totalLeakage, delta: overallRenewed - overallUp, byCarrier: carrierArray, byMonth: monthlyRetention }, leakage: { cancellations: cancTotal, cancellationCount: cancCount, nonRenewals: nrTotal, nonRenewalCount: nrCount, totalLeakage: totalLeakage, totalCount: cancCount + nrCount, pctOfBob: leakagePctBob, liveBob: bobTotal }, lossBudget: { budget: lossBudget, used: totalLeakage, remaining: lossBudget - totalLeakage, pctUsed: lossBudget > 0 ? Math.round((totalLeakage / lossBudget) * 1000) / 10 : 0 } } }; } catch (err) { return { ok: false, error: err.toString() }; } }
